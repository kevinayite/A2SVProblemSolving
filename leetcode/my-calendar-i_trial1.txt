from sortedcontainers import SortedList
class MyCalendar:

    def __init__(self):
        self.calendar = SortedList()

    def book(self, startTime: int, endTime: int) -> bool:
        index = self.calendar.bisect_right((startTime, endTime))
        if (index > 0 and self.calendar[index - 1][1] > startTime) or (index < len(self.calendar) and self.calendar[index][0] < endTime):
            return False
        self.calendar.add((startTime,  endTime))
        return True
