class Solution:
    def minMovesToSeat(self, seats: List[int], students: List[int]) -> int:
        seats.sort()
        students.sort()
        count = 0
        i = 0
        while i < len(seats) :
            if seats[i] != students[i]:
                diff = abs(seats[i] - students[i])
                count += diff
            i +=1
        return count
