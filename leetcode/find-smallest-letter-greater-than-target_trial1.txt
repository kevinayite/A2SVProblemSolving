class Solution:
    def nextGreatestLetter(self, letters: List[str], target: str) -> str:
        low = 0
        high = len(letters) - 1
        answer = ""
        while low <= high :
            mid = (low + high) // 2
            if letters[mid] > target:
                answer = letters[mid]
                high = mid - 1              
            else:
                low = mid + 1
        if answer:
            return answer
        else:
            return letters[0]
