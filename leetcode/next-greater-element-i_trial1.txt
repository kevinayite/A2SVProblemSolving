class Solution:
    def nextGreaterElement(self, nums1: List[int], nums2: List[int]) -> List[int]:
        stack = []    
        my_dict = {}  
        
        for num in nums2:
            # While the stack isn't empty AND the current number is bigger than 
            # the last number we put on the stack...
            while stack and num > stack[-1]:
                # we have found the 'next greater element' for the top of the stack.
                # Pop it and save the pair in our dictionary.
                my_dict[stack.pop()] = num            
            # Always add the current number to the stack to find its own next greater neighbor later
            stack.append(num)
        
        # Any numbers left in the stack have no greater element to their right
        for num in stack:
            my_dict[num] = -1
            
        return [my_dict[num] for num in nums1]
