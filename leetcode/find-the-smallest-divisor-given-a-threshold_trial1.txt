import math
class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        low = 1
        high = max(nums)
        answer = high 
        def calculate_sum(divisor):
            total_sum = 0
            for num in nums:
                total_sum += math.ceil(num / divisor)
            return total_sum

        while low <= high:
            mid = low + (high - low) // 2
            if calculate_sum(mid) <= threshold:
                answer = mid
                high = mid - 1
            else:
                low = mid + 1
        return answer
