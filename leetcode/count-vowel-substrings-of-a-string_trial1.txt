class Solution:
    def countVowelSubstrings(self, word: str) -> int:
        vowels = set("aeiou")
        count = 0
        for i in range(len(word)):
            freq = dict()
            for j in range(i, len(word)):
                if word[j] not in vowels: # not a vowel
                    break
                freq[word[j]] = freq.get(word[j], 0) + 1
                if len(freq) == 5: # as we are using dictionary when we reach 5 it is only composed of vowels aeiou
                    count +=1
        return count
