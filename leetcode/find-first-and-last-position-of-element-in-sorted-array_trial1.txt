class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:
        
        def search(t):
            low, high = 0, len(nums)
            while low < high:
                mid = (low + high) // 2
                if nums[mid] < t:
                    low = mid + 1
                else:
                    high = mid  # NOT mid - 1
            return low
        
        first = search(target)
        last = search(target + 1) - 1
        
        if first < len(nums) and nums[first] == target:
            return [first, last]
        return [-1, -1]
