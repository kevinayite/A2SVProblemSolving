class Solution:
    def largestRectangleArea(self, heights: List[int]) -> int:
        stack = []
        max_area = 0
        heights.append(0)

        for i, cur_height in enumerate(heights): 
            while stack and heights[stack[-1]] > cur_height:
                local_max_height = heights[stack.pop()]
                width = 0
                if not stack:
                    width = i
                else:
                    width = i - stack[-1] - 1
                max_area = max(max_area, local_max_height * width) 
            stack.append(i)
        return max_area
