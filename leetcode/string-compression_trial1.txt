from collections import Counter
class Solution:
    def compress(self, chars: List[str]) -> int:
        i = 0
        answer = 0
        while i < len(chars):
            group_length = 1 # lenght of repeated characters
            while (i + group_length < len(chars) and chars[i + group_length] == chars[i]):
                group_length += 1
            chars[answer] = chars[i] # here the case when the element appears once is already solved
            answer +=1
            if group_length > 1:
                str_repr = str(group_length)
                chars[answer: answer+len(str_repr)] = list(str_repr)
                answer += len(str_repr)
            i += group_length
        return answer
