# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def isPalindrome(self, head: Optional[ListNode]) -> bool:
        fast = slow = head
        while fast and fast.next:
            fast = fast.next.next
            slow = slow.next
        
        previous = None
        current = slow
        
        while current:
            next_temp = current.next
            current.next = previous
            previous = current
            current = next_temp
        
        left, right = head, previous
        while right:
            if left.val != right.val:
                return False
            left = left.next
            right = right.next
        return True
