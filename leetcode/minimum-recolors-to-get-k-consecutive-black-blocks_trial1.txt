from collections import Counter
class Solution:
    def minimumRecolors(self, blocks: str, k: int) -> int:
        left = 0
        result = float("inf")
        white_color = 0
        for right in range(len(blocks)):
            if blocks[right] == "W":
                white_color += 1
            if right - left + 1 == k:
                result = min(result, white_color)
                if blocks[left] == "W":
                    white_color -= 1
                left += 1
        return result
