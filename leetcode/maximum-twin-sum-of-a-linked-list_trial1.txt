# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def pairSum(self, head: Optional[ListNode]) -> int:
        fast = slow = head
        # reach the end of our linked list
        while fast and fast.next:
            fast = fast.next.next
            slow = slow.next
        previous = None
        current = slow
        while current:
            nxt = current.next
            current.next = previous
            previous = current 
            current = nxt
        max_sum = 0
        left, right = head, previous
        while right: # right start at the middle of the beginning of the  reversed second half
            max_sum = max(max_sum, left.val + right.val)
            left = left.next
            right = right.next
        return max_sum
