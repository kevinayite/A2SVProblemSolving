# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def splitListToParts(self, head: Optional[ListNode], k: int) -> List[Optional[ListNode]]:
        result = []
        n, current = 0, head
        # knowing the lenght
        while current:
            n += 1
            current = current.next
        base_size = n //k
        extra = n % k # handling cases
        current = head
        for i in range(k):
            part_head = current
            current_part_size = base_size + (1 if i < extra else 0) # if we have extra remaining distribute one, else add 0
            for j in range(current_part_size - 1):
                if current:
                    current = current.next
            if current:
                next_part_head = current.next
                current.next = None
                current = next_part_head
            result.append(part_head)
        return result
