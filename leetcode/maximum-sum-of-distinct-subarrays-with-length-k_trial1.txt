class Solution:
    def maximumSubarraySum(self, nums: List[int], k: int) -> int:
        left = 0
        count = {}
        max_sum = 0
        windows_sum = 0
        for right in range(len(nums)):
            num = nums[right]
            windows_sum += num
            count[num] = count.get(num, 0) + 1
            while count[num] > 1 or right - left +1 > k:
                left_num = nums[left]
                windows_sum -= left_num
                count[left_num] -= 1
                if count[left_num] ==0:
                    del count[left_num]
                left += 1
            if right - left + 1 == k:
                max_sum = max(max_sum, windows_sum)
        return max_sum
