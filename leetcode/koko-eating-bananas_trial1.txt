import math
class Solution:
    def minEatingSpeed(self, piles: List[int], h: int) -> int:
        def can_finish(k):
            hours = 0
            for p in piles:
                hours += (p + k - 1) // k  # ceil(p / k)
            return hours <= h

        low = 1
        high = max(piles)
        answer = high

        while low <= high:
            mid = (low + high) // 2

            if can_finish(mid):
                answer = mid
                high = mid - 1   # try smaller k
            else:
                low = mid + 1    # need bigger k

        return answer
