# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
# Definition for singly-linked list.

class Solution:
    def addTwoNumbers(self, l1: Optional[ListNode], l2: Optional[ListNode]) -> Optional[ListNode]:
        h1 = l1
        h2 = l2

        carry = 0
        ans = ListNode(0)
        cu = ans
        while h1 or h2 or carry:
            total = 0
            if h1:
                total += h1.val
                h1 = h1.next
            if h2:
                total += h2.val
                h2 = h2.next
            total += carry

            if total > 9:
                carry = 1
                total %= 10
            else:
                carry = 0
            cu.next = ListNode(total)
            cu = cu.next
        return ans.next
