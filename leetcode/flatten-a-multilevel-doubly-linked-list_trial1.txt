"""
# Definition for a Node.
class Node:
    def __init__(self, val, prev, next, child):
        self.val = val
        self.prev = prev
        self.next = next
        self.child = child
"""

class Solution:
    def flatten(self, head: 'Optional[Node]') -> 'Optional[Node]':
        current = head
        while current:
            if current.child:
                temp = current.next
                current.next = current.child
                current.child.prev = current
                current.child = None # redefine the pointer to child null
                tail = current.next # find the tail of the new child branch
                while tail.next:
                    tail = tail.next
                if temp:
                    tail.next = temp
                    temp.prev = tail
            current = current.next
        return head
