class Solution:
    def characterReplacement(self, s: str, k: int) -> int:
        count = {}
        left = 0
        max_freq = 0
        answer = 0
        for right in range(len(s)):
            count[s[right]] = count.get(s[right], 0) + 1 # Give me the value for s[right] if it exists, otherwise give me 0
            max_freq = max(max_freq, count[s[right]])

            while (right - left +1) - max_freq> k: #Number of replacements needed is too large So the window is invalid
                count[s[left]] -= 1 # Remove leftmost character Make window smaller until valid again
                left += 1
            answer = max(answer, right - left + 1) # Keep track of the largest valid window
        return answer
