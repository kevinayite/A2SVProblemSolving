class Solution:
    def maxDistance(self, position: List[int], m: int) -> int:
        position.sort()
        def canPlace(d):
            count = 1
            last = position[0]

            for i in range(1, len(position)):
                if position[i] - last >= d:
                    count += 1
                    last = position[i]
                if count >= m:
                    return True
            return False
        low, high = 1, position[-1] - position[0]
        answer = 0

        while low <= high:
            mid = low + (high - low) // 2
            if canPlace(mid):
                answer = mid
                low = mid + 1
            else:
                high = mid - 1 # This distance is too big, try smaller ones
        return answer
