class Solution:
    def countGood(self, nums: List[int], k: int) -> int:
        count = {}
        left = 0
        result = 0
        pairs = 0
        for right in range(len(nums)):
            pairs += count.get(nums[right], 0)
            count[nums[right]] = count.get(nums[right], 0) + 1
            # count = count[nums[right]]
            while pairs >= k:
                result += len(nums) -right
                count[nums[left]] -= 1
                pairs -= count[nums[left]] # here we remove the count of count because it form the amount of pairs
                left +=1
        return result
