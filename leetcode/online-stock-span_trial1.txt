class StockSpanner:

    def __init__(self):
        self.stack = [] # we will insert the combination of [price, span]
    def next(self, price: int) -> int:
        span = 1
        while self.stack and self.stack[-1][0] <= price: # comparing our last price in the stack with the current one
            span += self.stack.pop()[1] # to have the last span, but we are popping the pair from our stack
        self.stack.append([price, span])
        return span
