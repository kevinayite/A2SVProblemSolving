class Solution:
    def matchPlayersAndTrainers(self, players: List[int], trainers: List[int]) -> int:

        players.sort()
        trainers.sort()
        
        trainer_index = 0
        count = 0
        i =0 
        while i < len(players) and trainer_index < len(trainers):
            if players[i] <= trainers[trainer_index]:
                count += 1
                i +=1
                trainer_index +=1
            else:
                trainer_index +=1
        return count
