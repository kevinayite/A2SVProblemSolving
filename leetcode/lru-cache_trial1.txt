class Node:
    def __init__ (self, key, val):
        self.key = key
        self.val = val
        self.previous = self.next = None

class LRUCache:

    def __init__(self, capacity: int):
        self.capacity = capacity
        self.my_dict = {} # map key to node

        # let's have 2 pointers:
        # left pointer will help us to reach least recent used
        # right will help us to reach the most recent used
        self.left = Node(0,0) # we have 2 parameters for nodes
        self.right = Node(0, 0)
        self.left.next = self.right 
        self.right.previous = self.left # linking our 2 pointers
    
    def insert(self, node):
        previous = self.right.previous
        nxt = self.right
        previous.next = nxt.previous = node
        node.next = nxt
        node.previous = previous

    def remove(self, node):
        previous = node.previous
        nxt = node.next
        previous.next = nxt
        nxt.previous = previous

        
    def get(self, key: int) -> int:
        # If accessed, move it to the MRU position (right)
        if key in self.my_dict:
            self.remove(self.my_dict[key]) # we are removing it 
            self.insert(self.my_dict[key])
            return self.my_dict[key].val
        return -1
    
        

    def put(self, key: int, value: int) -> None:
        # If key exists, remove the old version from the list
        if key in self.my_dict: 
            self.remove(self.my_dict[key])
        self.my_dict[key] = Node(key, value)
        self.insert(self.my_dict[key])
        # Create/Update the node and add it to MRU position
        if len(self.my_dict) > self.capacity:
            # remove from the list and delete the least recent used from the hashmap as well
            lru = self.left.next
            self.remove(lru)
            del self.my_dict[lru.key]
