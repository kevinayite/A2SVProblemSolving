import math
class Solution:
    def dividePlayers(self, skill: List[int]) -> int:
        left = 0
        right = len(skill) -1
        skill.sort()
        target = skill[left] + skill[right]
        chemistry = 0
        while left < right:
            if skill[left] + skill[right] != target:
                return -1
            chemistry += skill[left] * skill[right]
            left += 1
            right -= 1
        return chemistry
