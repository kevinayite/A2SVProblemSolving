class Solution:
    def mySqrt(self, x: int) -> int:
        if x < 2:
            return x
        
        low, high = 1, x // 2
        ans = 0
        
        while low <= high:
            mid = low + (high - low) // 2
            if mid <= x // mid: # to avoid integer overflow
                ans = mid
                low = mid + 1
            else:
                high = mid - 1
        
        return ans
